"""
dag_order_simulator.py
──────────────────────
DAG giả lập đơn hàng Highlands Coffee theo thời tiết hiện tại => cần load thời tiết hiện tại từ DB và master data từ DB trước khi tạo đơn
Disclaimer: Không sửa masterdata randomly lắm. Nếu mà sửa profile khách hàng/trạng thái active của store, product thì lại phải handle historical data của khách, stores, prouducts để phản ánh đúng trạng thái đơn, nếu ko nó đang link trực tiếp thì đơn cũng bị thay đổi nếu master data thay đổi => schema ở highlands bị thay đổi

Schedule: mỗi 10 phút — đủ dày để có data CDC nhưng không spam DB.
Mỗi lần chạy tạo ra một batch đơn cho tất cả stores.

Tại sao là DAG thay vì script while-True?
  - Airflow quản lý retry, log, alert → không cần tự viết
  - Có thể trigger thủ công từ UI để test
  - Dừng/bật bằng toggle trên UI, không cần ssh vào container
"""

import random
import uuid

from airflow.sdk import dag, task
from airflow.providers.microsoft.mssql.hooks.mssql import MsSqlHook
from datetime import datetime
from sqlalchemy import text

# ─── CONFIG ───────────────────────────────────────────────────────────────────

_MSSQL_CONN_ID   = "mssql_highlands"   # Connection riêng trỏ vào HighlandsDB
_ORDERS_PER_STORE = (5, 8)             # Số đơn mỗi store mỗi lần chạy


def _get_engine():
    return MsSqlHook(mssql_conn_id=_MSSQL_CONN_ID).get_sqlalchemy_engine()


# ─── WEATHER PROFILES ─────────────────────────────────────────────────────────
# Nếu mưa thì khách mua delivery tăng + hot drink tăng. Ngược lại trời đẹp thì dine-in tăng, cold drink tăng

WEATHER_PROFILES = {
    "Rain":         {"delivery_prob": 0.80, "category_weights": {"Hot Drink": 4, "Cold Drink": 1, "Food": 2}},
    "Thunderstorm": {"delivery_prob": 0.90, "category_weights": {"Hot Drink": 5, "Cold Drink": 1, "Food": 3}},
    "Drizzle":      {"delivery_prob": 0.65, "category_weights": {"Hot Drink": 3, "Cold Drink": 1, "Food": 2}},
    "Clear":        {"delivery_prob": 0.25, "category_weights": {"Hot Drink": 1, "Cold Drink": 4, "Food": 1}},
    "Clouds":       {"delivery_prob": 0.45, "category_weights": {"Hot Drink": 1, "Cold Drink": 2, "Food": 1}},
}
_DEFAULT_PROFILE = {"delivery_prob": 0.40, "category_weights": {"Hot Drink": 1, "Cold Drink": 2, "Food": 1}}


# ─── DAG ──────────────────────────────────────────────────────────────────────

@dag(
    start_date=datetime(2026, 4, 26),
    schedule="*/10 * * * *",  # mỗi 10 phút — cron syntax: phút/10 * * * *
    catchup=False,             # không chạy bù các khoảng thời gian đã qua
    tags=["highlands", "simulator", "orders"],
)
def order_simulator_dag():

    @task
    def load_master_data() -> dict:
        """
        Load Stores, Customers, Products vào dict để truyền sang task tiếp theo.

        Trả về dict thay vì dataclass vì Airflow serialize dữ liệu giữa các task
        qua XCom (một dạng key-value store trong DB Airflow) — chỉ hỗ trợ
        JSON-serializable types: dict, list, str, int, float.
        """
        engine = _get_engine()
        with engine.connect() as conn: # bắt đầu một transaction
            stores = [                 
                {"store_id": r.store_id, "city_id": r.city_id}
                for r in conn.execute(
                    text("SELECT store_id, city_id FROM Stores WHERE is_active = 1")
                ).fetchall() 

                # get danh sách các active stores và city_id tương ứng của store
                # list(dict)
                # stores = [{"store_id": 1, "city_id": 1}, 
                #           {"store_id": 1, "city_id": 2}] 
            ]

            customer_ids = [
                r.customer_id
                for r in conn.execute(text("SELECT customer_id FROM Customers")).fetchall()

                # customer_ids = [1, 2, 3]
            ]

            products = [
                {"product_id": r.product_id, "category": r.category, "price": float(r.price)}
                for r in conn.execute(
                    text("SELECT product_id, category, price FROM Products WHERE is_active = 1")
                ).fetchall()

                # products = [{"product_id": 1, "category": "Hot Drink", "price": 40000},
                #             {"product_id": 2, "category": "Cold Drink", "price": 55000}]
            ]

        print(f"Loaded: {len(stores)} stores, {len(customer_ids)} customers, {len(products)} products")
        return {"stores": stores, "customer_ids": customer_ids, "products": products}
        # Xcom sau khi chạy load master data là đúng 1 cái dict key-value

    @task
    def fetch_weather() -> dict:
        """
        Đọc CurrentWeatherState để biết thời tiết mỗi thành phố.
        Trả về dict city_id (string) → weather info.

        Lưu ý: XCom serialize key thành string → dùng str(city_id) làm key.
        """
        engine = _get_engine()
        with engine.connect() as conn:
            rows = conn.execute(
                text("""
                    SELECT c.city_id, c.city_name, w.weather_condition, w.temperature
                    FROM   CurrentWeatherState w
                    JOIN   Cities c ON c.city_id = w.city_id
                """)
            ).fetchall()

        weather_map = {
            str(r.city_id): {
                "city_name": r.city_name,
                "condition": r.weather_condition or "Clouds",
                "temperature": float(r.temperature or 25.0)
                
                # Nếu r.weather_condition là None, "", 0, False → lấy "Clouds"
                # Nếu r.weather_condition có giá trị thật → lấy giá trị đó
                # tránh lấy NULL từ DB, generate đơn trung tính
            }
            for r in rows
        }
        print(f"Weather snapshot: {weather_map}")
        return weather_map
        # weather_map = {{"1": {"city_name": "Hanoi", "condition": "Cloud", "temperature": 25.0}},
        #                {"2": {"city_name": "Ho Chi Minh City", "condition": "Rain", "temperature": 19.7}}
        #               }

    @task
    def generate_and_insert_orders(master: dict, weather_map: dict) -> int:
        """
        Task cuối: tạo đơn hàng + INSERT trong 1 transaction.

        Nhận 2 tham số từ 2 task trước qua XCom —
        Airflow tự inject khi mày truyền return value của task vào đây.
        """
        stores       = master["stores"] # list các active stores
        customer_ids = master["customer_ids"]
        products     = master["products"]

        orders_to_insert = []

        """
        Với từng store trong active stores:
            1. Quyết định store này có bao nhiêu orders
            2. Với từng order trong orders (của từng store):
                - Quyết định số items
                - Với từng item thì quyết định product_id theo weights, quantity, tính subtotal
            <- ra thông tin của order đang xét, append vào list orders_to_insert của batch đang làm việc
        """

        for store in stores:
            # lấy weather từ weather_map ứng với city_id của store_id đang xét
            weather = weather_map.get(str(store["city_id"]))
            if not weather:
                print(f"Không có weather cho city_id={store['city_id']}, bỏ qua store {store['store_id']}")
                continue
            
            # lấy weather_profiles ứng với condition trên
            profile = WEATHER_PROFILES.get(weather["condition"], _DEFAULT_PROFILE) # handle không thấy loại condition
            n_orders = random.randint(*_ORDERS_PER_STORE) 

            for _ in range(n_orders):
                order_type = "Delivery" if random.random() < profile["delivery_prob"] else "Dine-in"
                customer_id = random.choice(customer_ids)

                # Tạo items — mỗi đơn 1-3 sản phẩm
                items = []
                total = 0.0
                for _ in range(random.randint(1, 3)): # random số items của orders này, mỗi lần chọn 1 item là 1 lần tung đồng xu vs xác suất (4 2 1) lấy ra k=1
                    # mỗi _ là một row của order_detail trong order đang xét
                    weights = [profile["category_weights"].get(p["category"], 1) for p in products]
                        # profile["category_weights"] = {"Hot Drink": 4, "Cold Drink": 1, "Food": 2}
                            # .get(p["category"], 1) nghĩa là ứng với category của từng product trong products thì lấy weight tương ứng với weather_profile đang xét, phải handle cả việc ko thấy category
                        # dict.get(key, default)
                        # Nếu key có trong dict → trả về value
                        # Nếu key KHÔNG có trong dict → trả về default thay vì crash, phòng trường hợp thêm category mới vào DB mà chưa update WEATHER_PROFILES thì sản phẩm đó vẫn có cơ hội được chọn thay vì weight = 0 never đc chọn
                    product = random.choices(products, weights=weights, k=1)[0]
                        # products = [{"product_id": 1, "category": "Hot Drink",  "price": 35000},
                        #             {"product_id": 2, "category": "Cold Drink",  "price": 45000},
                        #             {"product_id": 3, "category": "Food",        "price": 55000}
                        #            ]
                        # weights = [4, 1, 2]
                        # k=1 → chọn 1 phần tử → trả về LIST có 1 phần tử: [{"product_id": 1, ...}]
                        # [0] → lấy phần tử đầu ra khỏi list → dict thật sự
                    qty     = random.randint(1, 2)
                    subtotal = product["price"] * qty
                    total   += subtotal
                    items.append({
                        "detail_id":  str(uuid.uuid4()),
                        "product_id": product["product_id"],
                        "quantity":   qty,
                        "unit_price": product["price"],
                        "subtotal":   round(subtotal, 2),
                    })

                orders_to_insert.append({
                    "order_id":    str(uuid.uuid4()),
                    "store_id":    store["store_id"],
                    "customer_id": customer_id,
                    "order_type":  order_type,
                    "total_amount": round(total, 2),
                    "details":     items,
                    # chỉ để log, không INSERT
                    "_city":       weather["city_name"],
                    "_condition":  weather["condition"],
                    "_temp":       weather["temperature"],
                })

        # INSERT toàn bộ batch trong 1 transaction
        # Status bắt đầu là 'Pending' — giống đơn thật vừa đặt
        # Task advance_order_status sẽ UPDATE sang Preparing → Completed ở lần chạy sau
        # Mỗi UPDATE thay đổi updated_at → Datastream CDC bắt từng sự kiện riêng biệt
        engine = _get_engine()
        with engine.begin() as conn:
            for order in orders_to_insert:
                conn.execute(
                    text("""
                        INSERT INTO Orders
                            (order_id, store_id, customer_id, order_type,
                             status, total_amount, created_at, updated_at)
                        VALUES
                            (:order_id, :store_id, :customer_id, :order_type,
                             'Pending', :total_amount, GETDATE(), GETDATE())
                    """),
                    {k: v for k, v in order.items() if not k.startswith("_") and k != "details"},
                )
                    # order = {
                        # "order_id":     "abc-123",
                        # "store_id":     1,
                        # "customer_id":  2,
                        # "order_type":   "Delivery",
                        # "total_amount": 85000,
                        # "details":      [...],   # ← list items, không INSERT vào Orders
                        # "_city":        "Hanoi", # ← chỉ để log
                        # "_condition":   "Rain",  # ← chỉ để log
                        # "_temp":        22.5,    # ← chỉ để log
                    #   }

                    # Cần lọc ra chỉ các key dùng được trong SQL INSERT INTO Orders:
                    # Kết quả:
                        # {
                            # "order_id": "abc-123",
                            # "store_id": 1,
                            # "customer_id": 2,
                            # "order_type": "Delivery",
                            # "total_amount": 85000,
                        # }
                for detail in order["details"]:
                    conn.execute(
                        text("""
                            INSERT INTO OrderDetails
                                (detail_id, order_id, product_id, quantity, unit_price, subtotal)
                            VALUES
                                (:detail_id, :order_id, :product_id, :quantity, :unit_price, :subtotal)
                        """),
                        {**detail, "order_id": order["order_id"]},
                        # **detail là unpack toàn bộ detail ra, rồi thêm key "order_id" vào, vì cần order_id để biết detail này thuộc order_id nào
                    )


                print(
                    f"  [{order['_city']} | {order['_condition']} {order['_temp']:.1f}°C] "
                    f"{order['order_type']:8s} | {len(order['details'])} items | "
                    f"{order['total_amount']:,.0f} VND"
                )

                """ [Hanoi | Rain 22.5°C] Delivery  | 2 items |   85,000 VND
                    [Ho Chi Minh | Clear 31.0°C] Dine-in  | 1 items |   45,000 VND"""

        print(f"Inserted {len(orders_to_insert)} orders total.")
        return len(orders_to_insert)

    @task
    def advance_order_status() -> None:
        """
        Mô phỏng vòng đời đơn hàng thực tế — mỗi lần DAG chạy:

          Pending   (> 2 phút) → Preparing   [đang pha chế]
          Preparing (> 3 phút) → Completed   [giao xong]
          Pending   (random 5%) → Cancelled   [khách huỷ]

        Mỗi UPDATE thay đổi cột updated_at → Datastream CDC
        bắt từng sự kiện riêng biệt và đẩy lên GCS Bronze.
        Đây là lý do tại sao cần CDC thay vì chỉ snapshot cuối cùng.
        """
        engine = _get_engine()
        with engine.begin() as conn:
            # Pending → Preparing: đơn chờ > 2 phút
            r1 = conn.execute(text("""
                UPDATE Orders
                SET    status     = 'Preparing',
                       updated_at = GETDATE()
                WHERE  status = 'Pending'
                  AND  DATEDIFF(MINUTE, created_at, GETDATE()) > 2
            """))

            # Preparing → Completed: đang pha > 3 phút
            r2 = conn.execute(text("""
                UPDATE Orders
                SET    status     = 'Completed',
                       updated_at = GETDATE()
                WHERE  status = 'Preparing'
                  AND  DATEDIFF(MINUTE, updated_at, GETDATE()) > 3
            """))

            # Random cancel ~5% đơn Pending còn non (< 2 phút)
            # TABLESAMPLE lấy mẫu ngẫu nhiên ~5% rows ở storage level — nhanh, không scan toàn bảng
            r3 = conn.execute(text("""
                UPDATE Orders
                SET    status     = 'Cancelled',
                       updated_at = GETDATE()
                WHERE  status = 'Pending'
                  AND  DATEDIFF(MINUTE, created_at, GETDATE()) <= 2
                  AND  order_id IN (
                      SELECT TOP 5 PERCENT order_id
                      FROM   Orders
                      WHERE  status = 'Pending'
                      ORDER BY NEWID()  -- shuffle ngẫu nhiên trước khi lấy TOP 5%
                  )
            """))

        print(f"Advanced: {r1.rowcount} → Preparing, {r2.rowcount} → Completed, {r3.rowcount} → Cancelled")

    # ─── LUỒNG CHẠY ───────────────────────────────────────────────────────────
    #
    #  advance_order_status ─┐
    #                        │  (chờ cả 3 xong)
    #  load_master_data    ──┼─► generate_and_insert_orders
    #                        │
    #  fetch_weather       ──┘
    #
    # advance + load_master + fetch_weather chạy SONG SONG (không phụ thuộc nhau)
    # generate_and_insert_orders phụ thuộc cả 3 → chạy sau cùng

    advanced    = advance_order_status()
    master      = load_master_data()
    weather_map = fetch_weather()
    generate_and_insert_orders(master, weather_map) >> advanced  # đảm bảo advance chạy xong trước khi insert mới


order_simulator_dag()
